import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

class SimpleAnalytics:
    def __init__(self, file_path):
        self.file_path = file_path
        self.data = None

    def load_data(self):
        try:
            self.data = pd.read_csv(self.file_path)
            print("Data loaded successfully.")
        except FileNotFoundError:
            print("File not found.")
        except Exception as e:
            print(f"Error loading file: {e}")

    def show_info(self):
        if self.data is not None:
            print("\nData Information:")
            print(self.data.info())
        else:
            print("No data loaded.")

    def show_statistics(self):
        if self.data is not None:
            print("\nSummary Statistics:")
            print(self.data.describe())
        else:
            print("No data loaded.")

    def show_null_counts(self):
        if self.data is not None:
            print("\nMissing Values:")
            print(self.data.isnull().sum())
        else:
            print("No data loaded.")

    def plot_histogram(self, column):
        if self.data is not None and column in self.data.columns:
            plt.figure(figsize=(8, 5))
            sns.histplot(self.data[column].dropna(), kde=True)
            plt.title(f"Histogram of {column}")
            plt.xlabel(column)
            plt.ylabel("Frequency")
            plt.show()
        else:
            print(f"Column '{column}' not found or data not loaded.")

    def plot_scatter(self, x_col, y_col):
        if self.data is not None and x_col in self.data.columns and y_col in self.data.columns:
            plt.figure(figsize=(8, 5))
            sns.scatterplot(data=self.data, x=x_col, y=y_col)
            plt.title(f"Scatter Plot: {x_col} vs {y_col}")
            plt.show()
        else:
            print(f"Columns '{x_col}' or '{y_col}' not found or data not loaded.")
